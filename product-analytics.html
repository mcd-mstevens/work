Time to complete
Median minutes by monitor type, monthly
Median seconds by agent monitor type, monthly
-- MIXPANEL.PUBLIC.MP_MASTER_EVENT — funnel: start → save SELECT DATE_TRUNC('month', c.TIME) AS month, c.MONITOR_TYPE, MEDIAN(DATEDIFF('second', s.TIME, c.TIME)/60.0) AS median_minutes FROM MIXPANEL.PUBLIC.MP_MASTER_EVENT s JOIN MIXPANEL.PUBLIC.MP_MASTER_EVENT c ON s.DISTINCT_ID = c.DISTINCT_ID AND s.SESSION_ID = c.SESSION_ID WHERE s.MP_EVENT_NAME = 'Monitor Create Started' AND c.MP_EVENT_NAME = 'Monitor Created' AND c.MONITOR_TYPE != 'agent' GROUP BY 1, 2 ORDER BY 1
-- AI_AGENT.PUBLIC.AGENT_RUNS SELECT DATE_TRUNC('month', completed_at) AS month, AGENT_TYPE, MEDIAN(DATEDIFF('second', started_at, completed_at)) AS median_seconds FROM AI_AGENT.PUBLIC.AGENT_RUNS WHERE status = 'completed' GROUP BY 1, 2 ORDER BY 1
Completion rate
% of monitor creations completed, by type, monthly
% of agent monitor creations completed, by type, monthly
-- MIXPANEL.PUBLIC.MP_MASTER_EVENT — start → save funnel WITH starts AS ( SELECT DISTINCT_ID, SESSION_ID, MONITOR_TYPE, TIME FROM MIXPANEL.PUBLIC.MP_MASTER_EVENT WHERE MP_EVENT_NAME = 'Monitor Create Started'), saves AS ( SELECT DISTINCT_ID, SESSION_ID FROM MIXPANEL.PUBLIC.MP_MASTER_EVENT WHERE MP_EVENT_NAME = 'Monitor Created') SELECT DATE_TRUNC('month', s.TIME) AS month, s.MONITOR_TYPE, 100.0 * COUNT(c.DISTINCT_ID) / COUNT(s.DISTINCT_ID) AS pct FROM starts s LEFT JOIN saves c USING (DISTINCT_ID, SESSION_ID) GROUP BY 1, 2 ORDER BY 1
-- AI_AGENT.PUBLIC.AGENT_RUNS SELECT DATE_TRUNC('month', started_at) AS month, AGENT_TYPE, 100.0 * SUM(CASE WHEN status = 'completed' THEN 1 END) / COUNT(*) AS pct FROM AI_AGENT.PUBLIC.AGENT_RUNS GROUP BY 1, 2 ORDER BY 1
